import pandas as pd
import matplotlib as plt
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error, r2_score
import matplotlib.pyplot as plt
import seaborn as sns
In[35]:
house_price = pd.read_csv('train.csv')
In[39]:
x_axis = house_price[['GrLivArea','BedroomAbvGr','FullBath']]
y_axis = house_price['SalePrice']
In[40]:
X_train, X_test, y_train, y_test = train_test_split(x_axis, y_axis, test_size=0.2, random_state=42)
In[41]:
Predict_model = LinearRegression()
Predict_model.fit(X_train, y_train)
y_prediction = Predict_model.predict(x_axis)
residuals = y_axis - y_prediction
In[42]:
Mean_squared_error = mean_squared_error(y_axis,y_prediction)
R_squared = r2_score(y_axis,y_prediction)
print(f'Validation Mean Squared Error: {Mean_squared_error: 0.2f}')
print(f'Validation R-squared: {R_squared: 0.2f}')

Validation Mean Squared Error:  2635954158.51
Validation R-squared:  0.58
In[20]:
plt.figure(figsize=(10, 6))
sns.scatterplot(x=y_axis, y=y_prediction)
plt.title('Actual vs. Predicted Values (Validation Set)')
plt.xlabel('Actual Sale Price')
plt.ylabel('Predicted Sale Price')
plt.show()
In[43]:
plt.figure(figsize=(8, 5))

plt.show()
